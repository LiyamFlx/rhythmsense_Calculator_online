<script>

  function startUpload() {

    const fileInput = document.getElementById("fileInput");

    const file = fileInput.files[0];

    if (file) {

      document.getElementById("fileName").textContent = `Uploading: ${file.name}`;

      document.getElementById("progressContainer").style.display = "block";

      

      // Simulate upload progress

      let progress = 0;

      const progressBar = document.getElementById("progressBar");

      const interval = setInterval(() => {

        if (progress >= 100) {

          clearInterval(interval);

          document.getElementById("fileName").textContent = `Uploaded: ${file.name}`;

        } else {

          progress += 10;

          progressBar.style.width = `${progress}%`;

        }

      }, 300);

    }

  }


  function uploadFromUrl() {

    const url = document.getElementById("fileUrl").value;

    if (url) {

      document.getElementById("fileName").textContent = `Uploading from URL...`;

      document.getElementById("progressContainer").style.display = "block";


      // Simulate URL upload progress

      let progress = 0;

      const progressBar = document.getElementById("progressBar");

      const interval = setInterval(() => {

        if (progress >= 100) {

          clearInterval(interval);

          document.getElementById("fileName").textContent = `Uploaded from URL: ${url}`;

        } else {

          progress += 15;

          progressBar.style.width = `${progress}%`;

        }

      }, 300);

    } else {

      alert("Please enter a valid URL.");

    }

  }

</script>
